-- user Table Create SQL
CREATE TABLE user
(
    `user_id`     VARCHAR(20)    NOT NULL    COMMENT '아이디', 
    `user_pwd`    VARCHAR(20)    NOT NULL    COMMENT '비밀번호', 
    `user_name`   VARCHAR(10)    NOT NULL    COMMENT '이름', 
    `user_email`  VARCHAR(20)    NOT NULL    COMMENT '이메일', 
    `user_nick`   VARCHAR(10)    NOT NULL    COMMENT '닉네임', 
    `user_type`   VARCHAR(20)    NOT NULL    COMMENT '회원 타입', 
    PRIMARY KEY (user_id)
);

ALTER TABLE user COMMENT '회원';


-- user Table Create SQL
CREATE TABLE lecture
(
    `lec_num`          INT              NOT NULL    AUTO_INCREMENT COMMENT '강의 번호', 
    `lec_name`         VARCHAR(50)      NOT NULL    COMMENT '강의 이름', 
    `lec_strart_date`  DATETIME         NOT NULL    COMMENT '강의 시작 일자', 
    `lec_end_date`     DATETIME         NOT NULL    COMMENT '강의 종료 일자', 
    `user_id`          VARCHAR(20)      NOT NULL    COMMENT '강사 아이디', 
    `lec_content`      VARCHAR(1000)    NOT NULL    COMMENT '강의 내용', 
    PRIMARY KEY (lec_num)
);

ALTER TABLE lecture COMMENT '강의';


-- user Table Create SQL
CREATE TABLE lecture_user
(
    `lec_num`  INT            NOT NULL    COMMENT '강의 번호', 
    `user_id`  VARCHAR(20)    NOT NULL    COMMENT '아이디', 
    PRIMARY KEY (lec_num, user_id)
);

ALTER TABLE lecture_user COMMENT '수강중인 강좌 테이블';
